/* mount USB in type fat32 */
fdisk -l    # get the /dev/sd[a-z][ SECTOR ?=1 ]
mkdir /mnt/usb
mount -t vfat -o iocharset=utf8 /dev/sd[a-z][ SECTOR ?=1 ] /mnt/usb
/* unmount USB */
umount /mnt/usb
rm -rf /mnt/usb
