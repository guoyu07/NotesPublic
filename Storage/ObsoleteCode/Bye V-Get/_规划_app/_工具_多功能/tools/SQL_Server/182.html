获取表中记录的总行数，等－T-SQL系列

（1）、获取表中记录的总行数

SELECT COUNT(*) FROM seq_num

--更快的方法，只能取得表的总行数
SELECT ROWS
FROM sysindexes
WHERE id=OBJECT_ID("seq_num")
  AND indid<2
