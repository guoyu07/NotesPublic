create function GetMonths (@Month smallint, @Year int)
returns smallint
as
begin
 return(DAY(dateadd(dd,-1,DATEADD(m,1,cast(@Year as varchar(4)) + '-' + cast(@Month as varchar(2)) + '-01'))))
end

select DatabaseName.dbo.GetMonths (2,2002)
